* Early init (loaded before init.el)
Disable package.el at startup so straight.el is the only package manager; avoids the "straight.el was loaded when package.el was already loaded" warning.
#+begin_src emacs-lisp :tangle .emacs.d/early-init.el :mkdirp yes
(setq package-enable-at-startup nil)
#+end_src

* Bootstrap
** Straight.el
Bootstrap straight.el first — it is the sole package manager. Package.el is disabled in =early-init.el=.
#+begin_src emacs-lisp
  (defvar bootstrap-version)
  (let ((bootstrap-file
	 (expand-file-name
	  "straight/repos/straight.el/bootstrap.el"
	  (or (bound-and-true-p straight-base-dir)
	      user-emacs-directory)))
	(bootstrap-version 7))
    (unless (file-exists-p bootstrap-file)
      (with-current-buffer
	  (url-retrieve-synchronously
	   "https://raw.githubusercontent.com/radian-software/straight.el/develop/install.el"
	   'silent 'inhibit-cookies)
	(goto-char (point-max))
	(eval-print-last-sexp)))
    (load bootstrap-file nil 'nomessage))
#+end_src

* Auto-tangle on save

Automatically tangle Emacs.org and any file in emacs/ when saved.
#+begin_src emacs-lisp
  ;; Automatically tangle our Emacs.org config file when we save it
  (defun skls/org-babel-tangle-config ()
    (when (and (buffer-file-name)
               (or (locate-file "Emacs.org" (list (file-name-directory (buffer-file-name))))
                   (string-match-p "/emacs/.*\\.org\\'" (buffer-file-name))))
      (let ((org-confirm-babel-evaluate nil))
        (org-babel-tangle))))

  (add-hook 'org-mode-hook (lambda () (add-hook 'after-save-hook #'skls/org-babel-tangle-config)))
#+end_src
