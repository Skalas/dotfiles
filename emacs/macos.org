* macOS Setup
#+BEGIN_SRC emacs-lisp
  (when (equal system-type 'darwin)
    (set-face-attribute 'default nil :font "Fira Code" :height 110)

    (setq insert-directory-program "/opt/homebrew/bin/gls")
    (setq magit-git-executable "/opt/homebrew/bin/git")

    ;; Ensure PATH and exec-path include Homebrew and local bins
    (dolist (dir '("/opt/homebrew/bin/"
                  "/opt/homebrew/sbin/"
                  "/opt/homebrew/opt/libpq/bin/"
                  "/usr/local/bin"))
      (add-to-list 'exec-path dir))
    (add-to-list 'exec-path (expand-file-name "~/.local/bin/"))
    (add-to-list 'exec-path (expand-file-name "~/.pyenv/shims/"))
    (setenv "PATH" (string-join exec-path ":"))

    (unless (getenv "LANG") (setenv "LANG" "en_US.UTF-8"))
    (unless (getenv "LC_ALL") (setenv "LC_ALL" "en_US.UTF-8"))

    ;; Spanish keyboard special characters via Meta
    (global-set-key (kbd "M-3") (lambda () (interactive) (insert "#")))
